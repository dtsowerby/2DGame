/*
 * Animation System Usage Example:
 * 
 * 1. Create animations with frame arrays:
 *    unsigned int walkFrames[] = {1, 2, 3, 4};
 *    Animation walkAnim = createAnimation(walkFrames, 4, 8.0f, ANIMATION_LOOP);
 * 
 * 2. For simple 2-frame toggle animations (like the old pt system):
 *    Animation toggleAnim = createToggleAnimation(frame1, frame2, 5.0f);
 * 
 * 3. Set animation on entity:
 *    setEntityAnimation(&player, walkAnim);
 * 
 * 4. Update animations each frame:
 *    updateEntityAnimation(&player, state.deltaTime);
 * 
 * 5. The entity will automatically use the current animation frame when drawn
 * 
 * Animation Types:
 * - ANIMATION_LOOP: Repeats forever (good for idle, walk cycles)
 * - ANIMATION_ONCE: Plays once then stops (good for attacks, jumps)
 * - ANIMATION_PINGPONG: Plays forward then backward (good for breathing effects)
 */

/*
 * Particle System Usage Example:
 * 
 * 1. Create particle emitters:
 *    ParticleEmitter explosion = createExplosionEmitter(position, 50);
 *    ParticleEmitter smoke = createSmokeEmitter(position);
 * 
 * 2. Start/stop emitters:
 *    startEmitter(&smoke);      // Start continuous emission
 *    stopEmitter(&smoke);       // Stop emission
 *    emitBurst(&explosion, 30); // Emit 30 particles at once
 * 
 * 3. Update and render each frame:
 *    updateParticleEmitter(&emitter, deltaTime);
 *    renderParticleEmitter(&emitter, particleShaderProgram);
 * 
 * 4. Customize emitters:
 *    setEmitterColorRange(&emitter, startColor, endColor);
 *    setEmitterVelocityRange(&emitter, minVel, maxVel);
 *    setEmitterLifeRange(&emitter, minLife, maxLife);
 * 
 * 5. Clean up when done:
 *    destroyParticleEmitter(&emitter);
 * 
 * Particle Types:
 * - PARTICLE_LINEAR: Constant velocity particles
 * - PARTICLE_GRAVITY: Particles affected by gravity
 * - PARTICLE_RADIAL: Particles moving outward from center
 * - PARTICLE_ORBIT: Particles orbiting around a point
 * 
 * Preset Emitters:
 * - createExplosionEmitter(): Yellow to red explosion burst
 * - createSmokeEmitter(): Gray smoke rising upward
 * - createSparkEmitter(): Yellow/orange sparks with gravity
 * - createFireEmitter(): Red/yellow fire effect
 * - createRainEmitter(): Blue rain falling down
 */
